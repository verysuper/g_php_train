SQL語法(建立student資料表):
CREATE TABLE  `student` (
 `studno` VARCHAR( 10 ) NOT NULL COMMENT  '學號',
 `name` CHAR( 10 ) NOT NULL DEFAULT  'noname' COMMENT  '姓名',
 `phone` CHAR( 13 ) NULL COMMENT  '電話',
 `address` VARCHAR( 60 ) NULL COMMENT  '地址',
PRIMARY KEY (  `studno` )
) ENGINE = MYISAM ;
--------------------------------------------------------------------------------------
SQL語法(清空student資料表):
